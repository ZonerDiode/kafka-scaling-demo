## Start it
docker compose up -d

## Verify brokers
docker exec -it kafka1 kafka-broker-api-versions --bootstrap-server kafka1:9092

## Create a topic
docker exec -it kafka1 kafka-topics `
  --create `
  --topic events.raw `
  --partitions 1 `
  --replication-factor 3 `
  --bootstrap-server kafka1:9092

## Open the UI
http://localhost:8080
